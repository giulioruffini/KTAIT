% ── ABSTRACT ──────────────────────────────────────────────────────────
\begin{abstract}
\noindent We are building \lean{KTAIT}, a Lean~4 + Mathlib formalization whose purpose is
narrow and load-bearing: to \emph{machine-check that the Kolmogorov-Theory (KT) corollaries
follow logically from an explicit algorithmic-information-theory (AIT) interface}, and that
the KT ontology is typed so that structural errors (e.g.\ comparing a whole pattern to its own
part) cannot silently compile. We do \emph{not} re-prove classical AIT: the standard AIT and
probability results (the invariance theorem, the coding theorem, symmetry of information,
Kraft's inequality, Bayes' rule, the Solomonoff--Levin universal semimeasure, the recursion
theorem, Rice's and Chaitin's theorems, and the Vereshchagin--Vitányi structure-function
results) are taken as a clearly delimited \emph{axiom layer}. Every \emph{KT} corollary is then
\emph{proved} from that layer with no \lean{sorry}, and a toy model witnesses that the axiom
layer is jointly satisfiable (so the corollaries are not vacuously true). This working paper
states the methodology, fixes the axiom layer, inventories every theorem/lemma across the KT
corpus with its current formalization status, and sets a prioritized roadmap.
\end{abstract}

\vspace{0.5em}
\noindent\textbf{Keywords:} Kolmogorov Theory, algorithmic information theory, Lean~4, Mathlib,
formal verification, algorithmic regulator theorem, persistence, self-model.

\noindent\textbf{Classification:} MSC 68Q30, 03B35, 68V15 \quad/\quad ACM F.4.1

\section{Introduction and methodology}
\label{sec:intro}
% ══════════════════════════════════════════════════════════════════════

KT has matured to the point where the dominant risk is no longer in the classical results it
invokes, but in whether the \emph{new} KT corollaries are \emph{correctly typed} and
\emph{logically entailed} by those results. This project addresses exactly that risk with a
proof assistant.\cite{leanmathlib}

\subsection{The one rule}
\begin{thmbox}[The one rule]
Standard AIT (and probability) theorems are allowed to be \emph{axioms}. KT corollaries must be
\emph{proved} from those axioms with no \lean{sorry}.
\end{thmbox}
\noindent If a KT corollary needs a \lean{sorry}, that is a real gap in the paper. If a classical
AIT fact is an axiom, that is honest and expected: we are checking the \emph{dependency
structure}, not the AIT foundations.

\subsection{Why this is worth doing}
The bug class we most want to exclude is \emph{ontological}. For example, writing
$\Ik(A:\mathcal{S})$ where $\mathcal{S}\subset A$ is structurally vacuous (a whole versus its own
part); the correct object is \emph{temporal}, $\Ik(\mathcal{S}_t:\mathcal{S}_{t+\tau})$ or
$K(\mathcal{S}_{t+\tau}\mid \mathcal{S}_t,E,C)\ll K(\mathcal{S}_{t+\tau}\mid C)$. In \lean{KTAIT}
the KT roles (\emph{substrate, pattern, self-code, readout, regulator, time}) are \emph{distinct
types}; the only bridge to a common carrier is an explicit projection, so the bad version
\emph{fails to type-check}. The whole-vs-part error becomes a typed obligation, not a silent
mistake.

\subsection{Soundness: named hypotheses, not global axioms}
\label{sec:soundness}
A na\"ive encoding \lean{axiom F : forall (F : AITFrame), P F} asserting a law for \emph{every}
frame is \emph{inconsistent} here: one can hand-build a frame violating it (e.g.\ \lean{slack=0}
with mismatched $K$) and derive \lean{False}, which would make all corollaries vacuously true.
We therefore state each AIT fact as a \emph{named hypothesis} (a \lean{Prop} about a frame) and
let each corollary consume it. Consequently \lean{\#print axioms} on every KT corollary shows
only Lean's core axioms (\lean{propext}, \lean{Classical.choice}, \lean{Quot.sound}) --- strictly
stronger than ``rests on named AIT axioms.'' The toy model (\S\ref{sec:status}) then exhibits a
concrete frame satisfying the hypotheses, witnessing joint satisfiability.

\subsection{Scope}
The current effort is \textbf{Level~1} (a typed, axiomatic KT/AIT interface) plus
\textbf{Level~1.5} (the faithful \emph{probabilistic} ART, including its Bayesian posterior and
the coding-theorem bridge). Out of scope for now: a computability-backed $K$ from a concrete
universal machine (Level~2), and the differential-geometric (Lie-group / Noether) layer of the
world-models paper,\cite{ruffini2025worldmodels} which is a separate track.

\section{The standard AIT / probability axiom layer}
\label{sec:axioms}
% ══════════════════════════════════════════════════════════════════════

These are the results we \emph{assume}. They are classical and not in doubt; collecting them here
fixes precisely what the KT proofs are allowed to lean on. (Per K's directive, starting from these
as axioms --- and similarly for Bayes --- is acceptable and intended.)

\begin{defbox}[Axiom layer A --- complexity]
\textbf{Invariance theorem.} $K$ is machine-independent up to an additive $O(1)$.\\
\textbf{Coding theorem (unconditional).} $c_1 2^{-K(x)}\le m(x)\le c_2 2^{-K(x)}$, i.e.
$-\log_2 m(x)=K(x)\pm O(1)$.\\
\textbf{Coding theorem (conditional).} $m(x\mid y)\asymp 2^{-K(x\mid y)}$.\\
\textbf{Kraft--McMillan.} $\sum_p 2^{-|p|}\le 1$, hence $\sum_x m(x)\le 1$.\\
\textbf{Symmetry of information.} $\Ik(x{:}y)=K(x)+K(y)-K(x,y)=K(x)-K(x\mid y^\ast)+O(\log)$;
chain rule $K(x,y)=K(x)+K(y\mid x^\ast)+O(1)$.\\
\textbf{Multiplicity compression.} $N_{\le L}(x)\ge 2^r \Rightarrow K(x)\le L-r+O(\log L)$.
\end{defbox}

\begin{defbox}[Axiom layer B --- probability / Bayes]
\textbf{Solomonoff--Levin universal semimeasure.} $m(x)=\sum_{U(p)=x}2^{-|p|}$; conditional
$m(R\mid W)\asymp 2^{-K(R\mid W)}$.\\
\textbf{Bayes' rule + deterministic likelihood.} prior $P(p)=2^{-|p|}$, likelihood
$P(x\mid p)=\mathbf{1}\{U(p)=x\}$, evidence $P(x)=m(x)$.\\
\textbf{Excess-length decay.} $\Pr\{|p|\ge K(x)+k \mid x\}\le 2C\,2^{-k}$.
\end{defbox}

\begin{defbox}[Axiom layer C --- computability / logic]
\textbf{Kleene's second recursion theorem} (quines); \textbf{Rice's theorem};
\textbf{Chaitin's incompleteness} ($T$ proves no $K(x)>n$ for $n>c_T$; $K$ uncomputable);
\textbf{Vereshchagin--Vitányi}: the Kolmogorov structure function $h_x(\alpha)$ is uncomputable
and minimal sufficient statistics are not locatable.
\end{defbox}

\item \lean{boundary\_sufficient} --- WP0162 App.~E: the thin boundary as a compressed sufficient

\item \lean{low\_complexity\_shrinkage} --- ART Theorem~A4; \lean{CoarseGraining.theoremA},

\item \lean{Noether.trajLabel\_conserved}, \lean{Noether.conserved\_comp\_symm} --- the abstract
core of generalized Noether (Structured Dynamics paper): the trajectory label is conserved by the

of the orbit-label theorem.)
\item \lean{Homogeneous.homogeneousSpace} --- a transitive symmetry action makes the state space a
homogeneous space, $X \simeq G/H_x$ (Theorem A4's algebraic core; the manifold/dimension layer
remains future work).
\item \lean{Decoder.achievable\_computable\_of\_inverse} / \lean{Decoder.no\_universal\_decoder}
--- WP0058 Proposition~1: a \emph{total} computable inverse compiler that recognizes its own domain
would decide membership in the achievable set of the developmental map; hence none exists, and every
realized Lamarckian write-back channel is a partial map whose validity domain is fixed in advance.
Notable as the one KT corollary here that assumes \emph{no} AIT fact at all --- the engine
(\lean{achievable\_computable\_of\_inverse}) is unconditional, and only the non-vacuity of the

(\lean{ExistsUndecidableAchievableSet}). Satisfiability witness: \lean{inverts\_id}.
\item \lean{WriteBack.bandwidth\_le\_cond} and its two corollaries --- WP0058 Proposition~2. With
$\lambda_B := \Ik(a : H'\mid H)$ the write-back bandwidth, joint-dominates-marginal gives
$\lambda_B \le K(H'\mid H)+O(\log)$, in which \emph{no term involving the acquired state survives}.
The regimes differ only in what bounds $K(H'\mid H)$: the selection signal $K(\sigma)$ in the
Darwinian case (\lean{darwinian\_bandwidth\_le\_selection}, zeroth-order search), the decoder image
$K(C(a)\mid H)$ in the Lamarckian (\lean{lamarckian\_bandwidth\_le\_decoder\_image}, first-order).
Also \lean{decoder\_charged} ($K(C)\le K(H)+O(\log)$, Prop.~1's AIT half) and

however much was acquired). Witness \lean{ToyWB} satisfies both named AIT hypotheses \emph{and
attains} the Darwinian bound (\lean{toyWB\_selection\_bound\_tight}) --- the corollaries are tight,
not merely non-vacuous.
\item \lean{persistence\_conservation} / \lean{conservation\_tradeoff} --- WP0162 Proposition~2:
the ledger $K(O_W)=\Ik(O_W{:}R)+K(O_W\mid R^\ast)$ (from symmetry of information), and at fixed
$K(O_W)$, maximizing $\Ik$ $=$ minimizing the conditional residual.
\item \lean{meta\_persistence} / \lean{meta\_persistence\_limit} --- WP0162 \S6 (meta-persistence):
a collective submodel of stable complexity $k>0$ with a bounded transient
$K(S^C_t\mid(S^C_{t+\tau})^\ast)\le L$ (delivered by closure $+$ a Foster--Lyapunov contraction of
the constituent flow) has collective persistence $\mathrm{Pers}_C\ge 1-(L+\mathrm{slack})/k$, hence
$\to 1$ as $k\to\infty$ --- with \emph{no} shared or compatible objective among the parts (the
hypotheses mention only $S^C,k,L,\mathrm{slack}$). Non-vacuity: \lean{toy\_meta\_persistence\_fires}.
\item \lean{regulator\_selection} --- WP0162 Proposition~1: $\argmin_R K(R\mid W)=\argmax_R[M(W{:}R)-K(R)]$.
\item \lean{AITProb.probabilistic\_regulator\_theorem\_conditional} --- ART sharp form in the
$2^{-K(R\mid W)}$ reading (posterior favors regulators simple given the world).
\item \lean{PrefixMachine.lemma1\_posterior\_bounds} --- the Bayes$\leftrightarrow$Kolmogorov bridge,
$\tfrac{1}{c_2}2^{K(x)-|p|}\le P(p\mid x)\le \tfrac{1}{c_1}2^{K(x)-|p|}$.
\item \lean{self\_regulation\_temporal\_model} --- Proposition~3,
$K(S_{t+\tau}\mid S_t,E,C)\le K(S_{t+\tau}\mid C)-\Delta_{\rm self}+2\,\mathrm{slack}$.
\item \lean{pers\_eq\_nmai}, \lean{persistent\_pos} --- persistence as temporal self-information.
\end{itemize}

\section{Roadmap}
\label{sec:roadmap}
% ══════════════════════════════════════════════════════════════════════

Priorities are ordered to maximize KT-correctness assurance per unit effort, exploiting the fact
that the AIT/Bayes layer (\S\ref{sec:axioms}) may be assumed.

\begin{description}[leftmargin=1.4em,style=nextline]
\item[\textcolor{ForestGreen}{Phase~1 --- close the ART chain. \textbf{DONE.}}] The wrapper bound
Eq.~(6) is now a \emph{derived} theorem (\lean{wrapper\_bound}) from the coding theorem (\lean{CodingLB});
ART \textbf{Theorem~1} (\lean{theorem1\_posterior\_tilt}) and \textbf{Theorem~3}
(\lean{theorem3\_onoff\_evidence}, log-free form) are proved; \lean{probabilistic\_regulator\_theorem}
now consumes \lean{wrapper\_bound} and rests on the coding theorem $+$ Lemma~2. All four
\lean{\#print axioms} $=$ Lean core only.
\item[\textcolor{ForestGreen}{Phase~2 --- persistence conservation (WP0162 Prop.~2). \textbf{DONE.}}]
Ledger $K(O_W)=\Ik(O_W{:}R)+K(O_W\mid R^\ast)$ (\lean{persistence\_conservation}, from symmetry of
information) and the trade-off ``max $\Ik$ $=$ min conditional residual'' (\lean{conservation\_tradeoff}).
\item[\textcolor{ForestGreen}{Phase~3 --- regulator selection (WP0162 Prop.~1). \textbf{DONE.}}]
$\argmin_{R}K(R\mid W)=\argmax_R[M(W{:}R)-K(R)]$ proved (\lean{regulator\_selection}) from the
chain rule; the ART sharp form's $2^{-K(R\mid W)}$ reading is
\lean{probabilistic\_regulator\_theorem\_conditional}.
\item[\textcolor{ForestGreen}{Phase~4 --- self-model incompleteness. \textbf{DONE.}}]
The three independent obstructions: the quine floor (\lean{quine\_floor}, $|Q_A|\ge K(A)$),
the self-prediction dichotomy (\lean{self\_prediction\_dichotomy}, an axiom-free diagonalization),
and the Chaitin ceiling on minimality (\lean{chaitin\_blocks\_minimality}).
\item[\textcolor{ForestGreen}{Phase~5 --- coarse-graining uncomputability (WP0193). \textbf{DONE.}}]
Theorem~B and Corollary~B (regulatory coarse-graining uncomputable) proved by an abstract
reduction to Vereshchagin--Vitányi (\lean{theoremB}, \lean{corollaryB}).

\smallskip\noindent\textbf{All five roadmap phases are complete.} The WP0162 / WP0192 / WP0193
KT corollaries and the full probabilistic ART layer are machine-checked, \lean{sorry}-free,
resting only on Lean core plus the named AIT axiom layer of \S\ref{sec:axioms}.
\item[\textcolor{gray}{Later --- geometry track.}] The Lie-group/Noether layer of the world-models

is complete.
\end{description}
